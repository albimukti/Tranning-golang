CREATE TABLE IF NOT EXISTS public.url_mappings
(
    id integer NOT NULL DEFAULT nextval('url_mappings_id_seq'::regclass),
    short_url character varying(255) COLLATE pg_catalog."default" NOT NULL,
    long_url text COLLATE pg_catalog."default" NOT NULL,
    created_at timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT url_mappings_pkey PRIMARY KEY (id),
    CONSTRAINT url_mappings_short_url_key UNIQUE (short_url)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.url_mappings
    OWNER to postgres;
